add_tracegen(j9dmp.tdf)
add_library(j9dmp SHARED
	dmpagent.c
	dmpmap.c
	dmpqueue.c
	dmpsup.c
	FileStream.cpp
	heapdump.cpp
	heapdump_classic.c
	javadump.cpp
	
	#TODO:Only on zos
	#jobname.s
	TextFileStream.cpp
	trigger.c

	${CMAKE_CURRENT_BINARY_DIR}/ut_j9dmp.c
)

target_link_libraries(j9dmp
	PRIVATE
		j9vm_interface
		j9vm_gc_includes

		omrsig
		j9util
		j9utilcore
		j9avl
		j9hashtable
		j9thr
		j9pool
		j9zlib
		j9hookable
		j9zip
)

target_include_directories(j9dmp
	PRIVATE
		${CMAKE_CURRENT_BINARY_DIR}
		../shared_common
)

omr_add_exports(j9dmp
	JVM_OnUnload
	JVM_OnLoad
	J9VMDllMain
)

# Note we only want to grab the rasdump headers
# Parsing the debug info causes build errors
target_enable_ddr(j9dmp GLOB_HEADERS NO_DEBUG_INFO)
ddr_set_add_targets(j9ddr j9dmp)

install(
	TARGETS j9dmp
	LIBRARY DESTINATION ${j9vm_SOURCE_DIR}
	RUNTIME DESTINATION ${j9vm_SOURCE_DIR}
)
